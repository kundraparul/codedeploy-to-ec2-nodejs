#!/bin/bash
sudo apt-get -y update

# INSTALL NODEJS - ONLY FOR UBUNTU
sudo apt install nodejs -y

# INSTALL NPM & PM2 - ONLY FOR UBUNTU
sudo apt install npm -y
sudo npm install pm2@latest -g

# Go to your app folder (replace with actual path if needed)
cd /home/ubuntu/your-app-folder

# Start Node.js app with PM2
pm2 start server.js --name "server"

# Make PM2 restart app automatically on reboot
pm2 startup systemd -u ubuntu --hp /home/ubuntu
pm2 save
